1)

echo "Enter a value to convert feet to inch: "
read feet src_unit _ dest_unit
src_unit=Feet
dest_unit=Inch

case $src_unit-$dest_unit in
   Feet-Inch)
    n1=$(($feet*12))
    echo " The value of Inch is $n1 "  
    ;;
   *)
    echo "I don't know how to convert $src_unit to $dest_unit"  
    ;;
esac
